<script>

const apiUrl="https://code-hub-eta.vercel.app/api/userkey.js";

const loginUrl="https://code-hub-eta.vercel.app/API/v2/continuar-conta.html?token=SEU_TOKEN_AQUI";

function abrirLogin(){
window.open(loginUrl,"_blank");
}

async function buscarDados(userkey){

const res=await fetch(`${apiUrl}?userkey=${encodeURIComponent(userkey)}`);
const data=await res.json();

const nome=data.nome || "Não disponível";
const email=data.email || "Não disponível";
const uid=data.uid;

document.getElementById("usuario").innerHTML=`
<p><b>Nome:</b> ${nome}</p>
<p><b>Email:</b> ${email}</p>
<p><b>UID:</b> ${uid}</p>
`;

buscarProjetos(uid);

document.getElementById("loginArea").style.display="none";
document.getElementById("dadosUsuario").style.display="block";

}

async function buscarProjetos(uid){

const url=`https://corsproxy.io/?https://api.codehub.ct.ws/?uid=${uid}`;

const res=await fetch(url);
const projetos=await res.json();

let html="";

for(const projetoId in projetos){

const projeto=projetos[projetoId];

html+=`
<div class="projeto">

<b>Projeto:</b> ${projeto.name}<br>
<b>Slug:</b> ${projeto.slug}<br>

<b>Arquivos:</b>
`;

if(projeto.files){

for(const fileId in projeto.files){

const file=projeto.files[fileId];

html+=`
<div class="arquivo">

<b>${file.originalName || file.name}</b><br>

<button onclick='abrirArquivo(${JSON.stringify(file)})'>
Visualizar
</button>

<button onclick='window.open("${file.url || file.directUrl}","_blank")'>
Abrir externo
</button>

<div id="preview-${fileId}" class="preview"></div>

</div>
`;

}

}

html+=`</div>`;

}

document.getElementById("projetos").innerHTML=html;

}

function abrirArquivo(file){

const previewId="preview-"+file.id;
const container=document.getElementById(previewId);

const url=file.url || file.directUrl;

/* arquivos de código */

if(file.content){

let type="text/plain";

if(file.language==="html") type="text/html";
if(file.language==="css") type="text/css";
if(file.language==="javascript") type="application/javascript";
if(file.language==="json") type="application/json";

const blob=new Blob([file.content],{type:type});

const blobUrl=URL.createObjectURL(blob);

container.innerHTML=`<iframe src="${blobUrl}"></iframe>`;

}

/* mídia */

else{

container.innerHTML=`<iframe src="${url}"></iframe>`;

}

}

function sair(){

localStorage.removeItem("userkey");

window.history.pushState({}, "", window.location.pathname);

document.getElementById("loginArea").style.display="block";
document.getElementById("dadosUsuario").style.display="none";

}

const params=new URLSearchParams(window.location.search);

const userkeyUrl=params.get("userKey");
const userkeySalvo=localStorage.getItem("userkey");

if(userkeyUrl){

localStorage.setItem("userkey",userkeyUrl);

buscarDados(userkeyUrl);

window.history.pushState({}, "", window.location.pathname);

}
else if(userkeySalvo){

buscarDados(userkeySalvo);

}

📖 Funcionalidades do Sistema

1️⃣ Autenticação

  • Login via UserKey com redirecionamento automático
  • Salvamento automático do UserKey no localStorage
  • Remoção do UserKey da URL por segurança
  • Persistência de sessão entre recarregamentos

2️⃣ Visualização de Projetos

  • Lista todos os projetos do usuário autenticado
  • Exibe nome e slug de cada projeto
  • Lista todos os arquivos dentro de cada projeto

3️⃣ Visualização de Arquivos

  • Preview inline: Visualiza arquivos diretamente no iframe
  • Suporte a múltiplos formatos: HTML, CSS, JavaScript, JSON, imagens
  • Abrir externo: Abre o arquivo em nova aba
  • Blob URLs: Cria URLs temporárias para conteúdo de código
🔑 Endpoints da API:
https://code-hub-eta.vercel.app/api/userkey.js - Busca dados do usuário
https://api.codehub.ct.ws/?uid={UID} - Busca projetos do usuário

🚀 Como Usar

Passo 1: Obtenha seu token em https://alexande7888.github.io/API/criar.app
Passo 2: Substitua "SEU_TOKEN_AQUI" no código pelo seu token
Passo 3: Copie o código completo e cole em um arquivo HTML
Passo 4: Abra no navegador e faça login
Passo 5: Visualize seus projetos e arquivos!
